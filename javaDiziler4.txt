package javaapplication1;
import java.util.Scanner;
import java.util.Arrays;

public class dizilerGelismis {
    public static void main(String[] args) {
        Scanner girdi = new Scanner(System.in);
        
        System.out.print("Lütfen dizinin eleman sayısını belirtin: ");
        int n = girdi.nextInt();
        
        int[] dizi = new int[n];
        
        System.out.println("Lütfen dizinin elemanlarını girin: ");
        for (int i = 0; i < n; i++) {
            if (girdi.hasNextInt()) {
                dizi[i] = girdi.nextInt();
            }
        }
        
        System.out.print("Bulmak istediğiniz eleman: ");
        int aranan = girdi.nextInt();
        
        int arananIndex = Arrays.binarySearch(dizi, aranan);
        
        if(arananIndex >= 0){
            System.out.println("Aradığınız sayı dizide " + arananIndex + " numaralı indiste bulunuyor.");
        }else{
            System.out.println("Aradığınız sayı dizide bulunmuyor.");
        }
        
        System.out.println(" ");
        girdi.close();
    }
}
